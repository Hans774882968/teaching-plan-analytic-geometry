（选自 2023 杨春湖实验游园真题（一））甲、乙、丙三个人抽红包，一共有九个红包，大、中、小各三个。甲抽到了四个红包，大、中、小红包至少一个，总共 50 元，乙抽到了 3 个红包，总共也 50 元，丙抽到 2 个红包，总共 14 元，问三个红包的钱的分别是多少？

假设3人是按顺序不放回抽取，那么全部加起来除以3，得大+中+小=114/3=38。于是甲除了大中小以外还抽到一个12元的红包。12小于38/3所以只能是中或小红包。

然后看丙的两个红包，它有两中、两小、一中一小，3种可能。 两中的话，甲的12就是大红包了，不可能。 一中一小的话，小=14-12=2，大=38-14=24，那么甲额外一个中，丙一中一小，乙就只能2大1小， $24*2+2=50$ ，ok。 2小的话，2小=14，小=7，中=12，大=19，这时乙只能是2大1中， $19*2+12=50$ ，ok。

那就有两个解。如果没有上面的假设，那就写代码算。

```js
function solveRedEnvelope() {
  const solutions = [];
  for (let S = 1; S <= 13; S++) {
    for (let M = S + 1; M <= 49; M++) {
      for (let L = M + 1; L <= 50; L++) {
        const c_combinations = [
          [2, 0, 0], [0, 2, 0], [0, 0, 2], [1, 1, 0], [1, 0, 1], [0, 1, 1],
        ];
        const c_valid = c_combinations.some(([a, b, c]) => a * L + b * M + c * S === 14);
        if (!c_valid) continue;

        const b_combinations = [
          [3, 0, 0], [2, 1, 0], [2, 0, 1], [1, 2, 0], [1, 1, 1],
          [1, 0, 2], [0, 3, 0], [0, 2, 1], [0, 1, 2], [0, 0, 3],
        ];
        const b_valid = b_combinations.some(([a, b, c]) => a * L + b * M + c * S === 50);
        if (!b_valid) continue;

        const a_combinations = [
          [2, 1, 1], [1, 2, 1], [1, 1, 2],
        ];
        const a_valid = a_combinations.some(([a, b, c]) => a * L + b * M + c * S === 50);
        if (!a_valid) continue;

        const aComb = a_combinations.findIndex(([a, b, c]) => a * L + b * M + c * S === 50);
        const bComb = b_combinations.findIndex(([a, b, c]) => a * L + b * M + c * S === 50);
        const cComb = c_combinations.findIndex(([a, b, c]) => a * L + b * M + c * S === 14);

        solutions.push({ L, M, S, aComb: a_combinations[aComb], bComb: b_combinations[bComb], cComb: c_combinations[cComb] });
      }
    }
  }
  return solutions;
}

const results = solveRedEnvelope();
results.forEach((sol, i) => {
  console.log(`大红包: ${sol.L}, 中红包: ${sol.M}, 小红包: ${sol.S}, a的组合: [${sol.aComb}], b的组合: [${sol.bComb}], c的组合: [${sol.cComb}]`);
});
```
